'use client';

import React, { useEffect, useRef, useState } from 'react';

// --- Marker function (can be outside component or memoized if needed) ---
const addMarker = (map, L, latlng, deviceName, status) => {
    let iconUrl;
    switch (status) {
        case 'Online': iconUrl = '/marker-green.png'; break;
        case 'Offline': iconUrl = '/marker-red.png'; break;
        case 'Warning': iconUrl = '/marker-yellow.png'; break;
        default: iconUrl = '/marker-blue.png'; break; // Default or other statuses
    }

    const markerIcon = L.icon({
        iconUrl: iconUrl,
        shadowUrl: '/marker-shadow.png', // Assuming you have a shadow image
        iconSize: [25, 41],
        iconAnchor: [12, 41],
        popupAnchor: [1, -34],
        shadowSize: [41, 41],
        shadowAnchor: [12, 41] // Often same as iconAnchor for default shadow
    });

    L.marker(latlng, { icon: markerIcon }).addTo(map)
        .bindPopup(`<b>${deviceName}</b><br>Status: ${status}`);
};
// --- End of addMarker function ---


function MapComponent({ layoutReady }) { // Add layoutReady prop
    const mapRef = useRef(null); // Holds the Leaflet map instance
    const mapContainerRef = useRef(null); // Ref for the container div <div id="mapid">
    const [isMapInitialized, setIsMapInitialized] = useState(false); // Track initialization
    const initTimeoutRef = useRef(null); // Ref to store timeout ID
    const resizeObserverRef = useRef(null);
    const isObserverActiveRef = useRef(false); // Ref to track if the observer is active

    useEffect(() => {
        console.log('MapComponent useEffect triggered.');

        // Ensure the container ref is attached and the layout is ready
        if (!mapContainerRef.current || !layoutReady) {
            console.log('Map container ref not available yet or layout not ready.');
            return;
        }

        const mapElement = mapContainerRef.current; // Use the ref directly
        let leafletInstance = null; // To store the dynamically imported L

        // --- ResizeObserver Logic ---
        const resizeObserver = new ResizeObserver(entries => {
            if (!isObserverActiveRef.current){
                return;
            }
            // We only observe one element, so entries[0] is fine
            const entry = entries[0];
            const { width, height } = entry.contentRect;

            console.log(`ResizeObserver: Container dimensions: ${width} x ${height}`);

            // Clear any pending initialization timeout from previous observer calls
            if (initTimeoutRef.current) {
                clearTimeout(initTimeoutRef.current);
                initTimeoutRef.current = null;
                console.log('Cleared previous init timeout.');
            }

            // Check dimensions and initialization status
            if (width > 0 && height > 0 && !mapRef.current && !isMapInitialized) {
                console.log('Container has dimensions, scheduling map initialization...');

                // --- Use setTimeout to delay initialization slightly ---
                initTimeoutRef.current = setTimeout(() => {
                    console.log('setTimeout callback: Attempting to initialize map...');
                    // Import Leaflet dynamically (inside timeout)
                    import('leaflet').then(L => {
                        leafletInstance = L; // Store L for potential cleanup if needed later
                        console.log('Leaflet imported successfully (in timeout):', L);

                        // Double-check mapRef and dimensions AGAIN just before init
                        // This check is crucial in case something changed during the timeout
                        if (mapElement.offsetWidth > 0 && mapElement.offsetHeight > 0 && !mapRef.current) {
                            try {
                                console.log('Initializing Leaflet map (in timeout)...');
                                mapRef.current = L.map(mapElement).setView([51.505, -0.09], 13);
                                console.log('Leaflet map instance created:', mapRef.current);

                                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                                    attribution: '© <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
                                }).addTo(mapRef.current);
                                console.log('Tile layer added.');

                                // Add Markers (using the function defined outside)
                                addMarker(mapRef.current, L, [51.505, -0.09], 'Device 1', 'Online');
                                addMarker(mapRef.current, L, [51.510, -0.10], 'Device 2', 'Offline');
                                addMarker(mapRef.current, L, [51.495, -0.08], 'Device 3', 'Warning');
                                console.log('Markers added.');

                                setIsMapInitialized(true); // Mark as initialized

                                // IMPORTANT: Stop observing once initialized
                                console.log('Map initialized, stopping ResizeObserver.');
                                resizeObserverRef.current.unobserve(mapElement);
                                isObserverActiveRef.current = false; // Disable observer
                            } catch (error) {
                                console.error('Error initializing Leaflet map (in timeout):', error);
                            }
                        } else {
                            console.log('Skipping initialization inside timeout (map ref exists or dimensions became zero/invalid again).');
                        }
                    }).catch(error => {
                        console.error('Failed to import Leaflet (in timeout):', error);
                    });
                }, 100); // Delay of 100 ms pushes it to the next event loop cycle
                // --- End of setTimeout ---

            } else if (width > 0 && height > 0 && (mapRef.current || isMapInitialized)) {
                console.log('Container resized, but map already initialized.');
            } else if (!mapRef.current && !isMapInitialized) {
                // Only log this if we haven't initialized yet
                console.log('Container still has zero dimensions.');
            } else {
                // Handles cases where map is initialized but dimensions somehow became zero
                console.log('Observer fired, but map is already initialized or dimensions are zero.');
            }
        });

        // Start observing the map container element
        console.log('Starting ResizeObserver for map container.');
        resizeObserverRef.current = resizeObserver;
        isObserverActiveRef.current = true; // Enable observer
        resizeObserver.observe(mapElement);

        // --- Cleanup Function ---
        return () => {
            console.log('MapComponent cleanup: Disconnecting ResizeObserver and removing map.');
            // Clear timeout on cleanup
            if (initTimeoutRef.current) {
                clearTimeout(initTimeoutRef.current);
            }
            if (resizeObserverRef.current) {
                resizeObserverRef.current.disconnect(); // Stop observing
                isObserverActiveRef.current = false; // Ensure observer is disabled
            }

            if (mapRef.current) {
                mapRef.current.remove(); // Clean up Leaflet instance
                mapRef.current = null;
            }
            setIsMapInitialized(false); // Reset initialization state
        };
    }, [layoutReady]); // Dependency array now includes layoutReady

    // Assign the ref to the div
    // Ensure necessary Tailwind classes for layout are present (w-full, h-full)
    // Remove temporary background color if map appears
    return <div id="mapid" ref={mapContainerRef} className="w-full h-full min-h-[400px] z-10" />;
}

export default MapComponent;

    { icon: 'fa-cog', label: 'Settings', path: '/settings' },
  ];

  // Base classes without fixed width
  const baseClasses = "flex flex-col h-full bg-neutral-50 dark:bg-neutral-900 shadow-lg transition-transform duration-300 ease-in-out flex-shrink-0"; // Added flex-shrink-0
  // Positioning for mobile/desktop toggle
  const positioningClasses = "fixed inset-y-0 left-0 z-50 md:static md:shadow-none md:translate-x-0";
  const visibilityClasses = isOpen ? "translate-x-0" : "-translate-x-full";

  // Dynamic style for width
  const dynamicWidthStyle = { width: `${width}px` };

  return (
    // Apply ref and dynamic width style here
    <div
      ref={ref}
      style={dynamicWidthStyle}
      className={`${baseClasses} ${positioningClasses} ${visibilityClasses}`}
      aria-hidden={!isOpen && typeof window !== 'undefined' && window.innerWidth < 768} // Hide when closed on mobile
      aria-modal={isOpen && typeof window !== 'undefined' && window.innerWidth < 768} // Indicate modal behavior on mobile
      role="navigation" // Add role
    >
      {/* Inner container for padding and overflow */}
      <div className="flex flex-col h-full p-4 space-y-2 overflow-y-auto">
        <div className="flex items-center justify-between mb-4">
          <h2 className="text-lg font-bold text-neutral-800 dark:text-neutral-100">Menu</h2>
          {/* Close button only visible on smaller screens where sidebar is overlay */}
          <button
            type="button"
            onClick={onClose}
            className="md:hidden text-neutral-600 dark:text-neutral-400 hover:text-neutral-800 dark:hover:text-neutral-200 focus:outline-none"
            aria-label="Close menu"
          >
            <i className="fas fa-times fa-lg" aria-hidden="true"></i>
          </button>
        </div>

        <ul className="flex-grow space-y-2">
          {menuItems.map((item) => (
            <li key={item.label}>
              <Link
                href={item.path}
                className="flex items-center p-2 space-x-3 transition-colors duration-200 rounded-md hover:bg-neutral-200 dark:hover:bg-neutral-700"
                aria-label={`Navigate to ${item.label}`}
                // Close sidebar on navigation click, especially useful on mobile
                onClick={onClose}
              >
                <i className={`fas ${item.icon} fa-fw text-neutral-600 dark:text-neutral-400`} aria-hidden="true"></i>
                <span className="font-medium text-neutral-800 dark:text-neutral-100">{item.label}</span>
              </Link>
            </li>
          ))}
        </ul>

        <div className="pt-4 mt-auto border-t border-neutral-200 dark:border-neutral-700">
          <p className="text-sm text-center text-neutral-500 dark:text-neutral-400">Version 1.0.0</p>
        </div>
      </div>
    </div>
  );
});

// Add display name for better debugging with forwardRef
Sidebar.displayName = 'Sidebar';

Sidebar.propTypes = {

  width: PropTypes.number.isRequired, // Add width prop type
